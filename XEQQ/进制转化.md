int To10(int m, string s){ //转10
    int ans=0;
    for(int i=0;i<s.size();i++){
        char t=s[i];
        if(t>='0'&&t<='9') 
			ans=ans*m+t-'0';
        else 
			ans=ans*m+t-'a'+10;
    }
    return ans;
}
